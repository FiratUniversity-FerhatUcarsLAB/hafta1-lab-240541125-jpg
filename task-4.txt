Fonksiyon LegoSekilOlustur(hedef_sekil):

    // 1. Şekli analiz et
    boyutlar = SekilBoyutlariniHesapla(hedef_sekil)
    gerekli_parcalar = GerekliParcalariHesapla(hedef_sekil, boyutlar)

    // 2. Parçaları temin et
    eldeki_parcalar = ParcaListesiniHazirla()
    
    Eğer gerekli_parcalar ⊄ eldeki_parcalar:
        Eksik parçaları raporla
        Geri dön: "İnşa edilemez"

    // 3. İnşa planını oluştur
    insa_adimlari = InsaaPlaniniOlustur(hedef_sekil, gerekli_parcalar)

    // 4. İnşaya başla
    Döngü her adım içinde insa_adimlari:
        ParçayiYerineKoy(adım.parca, adım.konum, adım.yon)
        InsaDurumunuGuncelle()

    // 5. Tamamlandığını bildir
    Geri dön: "Şekil başarıyla oluşturuldu"
